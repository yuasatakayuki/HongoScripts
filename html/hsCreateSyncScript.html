#!/bin/bash

#2013-11-26 Takayuki Yuasa

if [ _$1 = _ ]; then
echo "give sync destination hostname and path."
echo "e.g. username@server.abcdefg.com:backup/"
exit
fi

path=$1

scriptName=doSynchonize.sh

cat << EOF > $scriptName
echo "============================================"
echo "From . to $path"
echo "---------------------------------------------"
rsync -auv --progress . $path

echo "============================================"
echo "From $path to ."
echo "---------------------------------------------"
rsync -auv --progress $path .
EOF

chmod 755 $scriptName
