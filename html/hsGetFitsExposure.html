#!/bin/bash

#20090522 Takayuki Yuasa updated

if [ _$1 = _ ];
then
echo "exposure.shhsGetFitsExposure FILE (value)"
exit
fi

file=$1
tmp=`get_hash_random.plhsHashRandom`.fits
ln -s $file $tmp

if [ _$2 = _ ];
then

i=`fits_has_header_keyword.shhsFitsIsHeaderKeywordDefined $tmp EXPOSURE`

if [ _$i != _ ]; then
getheader.shhsFitsGetHeader $tmp $i exposure
fi

else

setheader.shhsFitsSetHeader $tmp 0 exposure $2
setheader.shhsFitsSetHeader $tmp 1 exposure $2
setheader.shhsFitsSetHeader $tmp 2 exposure $2

fi


rm -r $tmp
