if(ARGV.length==0)then
print "usage whitedwarf_surface_area_for_mass.rbhsWhiteDwarfSurfaceAreaForMass (white dwarf mass in units of Msun) (output unit {cm2 m2 km2 Asun Aearth};optional, default=cm2)\n"
exit -1
end

unit="cm2"

if(ARGV.length==2)then
unit=ARGV[1]
end

mwd=ARGV[0].to_f()
if(1.44<=mwd or mwd<=0)then
print "0\n"
exit -1
end

rwd=7.8e8*Math.sqrt((1.44/mwd)**(2.0/3.0)-(mwd/1.44)**(2.0/3.0)) #cm
awd=4*3.1415926535*rwd*rwd

awd_out=awd
if(unit=="cm2")then
 awd_out=awd
elsif(unit=="m2")then
 awd_out=awd/10000
elsif(unit=="km2")then
 awd_out=awd/100000/100000
elsif(unit=="Asun")then
 awd_out=awd/100000/(1392000)/100000/(1392000)
elsif(unit=="Aearth")then
 awd_out=awd/100000/6356/100000/6356
else
 awd_out=0
end

print "#{awd_out}\n"
