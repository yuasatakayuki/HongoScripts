use Math::Trig;

if(@ARGV!=2){
 print "usage angulardistance.plhsAstroCalcAngularDistance (kpc) (arcmin)\n";
 print "return : projected distance in a unit of kpc\n";
 exit;
}

$r=$ARGV[0];
$arcmin=$ARGV[1];

$degree=$arcmin/60;
$radian=$degree/360*2*pi;

$distance=$r*$radian;
print $distance."\n";
