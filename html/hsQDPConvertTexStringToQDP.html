#!/usr/bin/env ruby

#20090611 Takayuki Yuasa

if(ARGV.length==0)then
print "usage : convert_tex_style_line_to_qdp_style.rbhsQDPConvertTexStringToQDP (tex style text to be converted)\n"
exit
end

line=ARGV[0]

line=line.gsub(/\^\{(.*?)\}/,'\\u\1\\d')
line=line.gsub(/_\{(.*?)\}/,'\\d\1\\u')
print line
