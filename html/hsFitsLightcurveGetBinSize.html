#!/bin/bash

#20090306 Takayuki Yuasa

if [ _$1 = _ ];
then
echo "fits_lightcurve_get_binsize.shhsFitsLightcurveGetBinSize FILE"
exit
fi

tmp=`get_hash_random.plhsHashRandom`

nrows=`getheader.shhsFitsGetHeader $1 1 NAXIS2`
fits_get_cell_value.shhsFitsGetTableCellValue $1 1 TIME `calc $nrows-1`-$nrows > $tmp

perl << EOF
open(FILE,"$tmp");
@file=<FILE>;close FILE;
print \$file[1]-\$file[0];
EOF

rm $tmp
