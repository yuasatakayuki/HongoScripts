#!/usr/bin/env ruby

if (ARGV[2]==nil) then
print "give binFrom, binTo, nStep, and (optional) separator\n"
end

binFrom=ARGV[0].to_f
binTo=ARGV[1].to_f
nSteps=ARGV[2].to_i

separator="\n"
if(ARGV[3]!=nil)then
 separator=ARGV[3]
end

logBinFrom = Math.log10(binFrom)
logBinTo = Math.log10(binTo)
step = (logBinTo - logBinFrom) / nSteps

i=0

while i<nSteps+1 do
 print 10**(logBinFrom + step * i)
 if(i!=nSteps)then
  print separator
 end
 i=i+1
end
